### Router — CSR SPA

```tsx
// routes/posts.tsx
export const Route = createFileRoute("/posts")({
  // runs in the browser, cached + prefetched
  loader: () => fetchPosts(),
  component: Posts,
})
```

### Start — same route, plus a server

```tsx
// routes/posts.tsx
const getPosts = createServerFn().handler(
  () => db.query.posts.findMany() // server only
)

export const Route = createFileRoute("/posts")({
  // runs on the server first — SSR'd, streamed
  loader: () => getPosts(),
  component: Posts,
})
```
